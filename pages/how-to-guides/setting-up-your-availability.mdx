import { useEffect } from "react"

export const Redirect = () => {
  useEffect(() => {
    {/* Only perform the redirect on the published site (no need in the preview) */}
    if (window.location.pathname?.startsWith('/docs/how-to-guides/setting-up-your-availability')) {
      window.location.href = "https://cal.com/docs/how-to-guides/availability/setting-up-your-availability";
    }
  }, [])
  
  return <></>
}

<Redirect />
