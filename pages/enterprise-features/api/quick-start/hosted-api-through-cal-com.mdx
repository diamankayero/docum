import { useEffect } from "react"

export const Redirect = () => {
  useEffect(() => {
    {/* Only perform the redirect on the published site (no need in the preview) */}
    if (window.location.pathname?.startsWith('/docs/enterprise-features/api/quick-start/hosted-api-through-cal-com')) {
      window.location.href = "https://cal.com/docs/enterprise-features/api/quick-start";
    }
  }, [])
  
  return <></>
}

<Redirect />
